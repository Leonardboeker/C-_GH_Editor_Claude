// Template: Panel Labeler
// Inputs:  panels         -> List<Brep>    (List Access)  -- Breps from panel_volume_generator
//          outerCentroids -> List<Point3d>  (List Access)  -- from panel_volume_generator outerCentroids
//          gradient       -> List<Color>   (List Access)  -- 2+ colors; script interpolates
//          prefix         -> string        (Item Access)  -- label prefix e.g. "P_"
//          startIndex     -> int           (Item Access)  -- first number (default 1)
// Outputs: labeledPanels -> List<Brep>    -- Breps with name in UserDictionary["PartName"]
//          labelPoints   -> List<Point3d> -- outer face centroids (pass-through)
//          labelText     -> List<string>  -- full label per panel e.g. "P_01"
//          panelColors   -> List<Color>   -- interpolated color per panel
//          names         -> List<string>  -- same as labelText (parallel output)

=== USING STATEMENTS ===

using System;
using System.Collections;
using System.Collections.Generic;
using System.Drawing;

using Rhino;
using Rhino.Geometry;

using Grasshopper;
using Grasshopper.Kernel;
using Grasshopper.Kernel.Data;
using Grasshopper.Kernel.Types;

=== CLASS AND RUNSCRIPT SIGNATURE ===

public class Script_Instance : GH_ScriptInstance
{
  private void RunScript(
    List<Brep>    panels,
    List<Point3d> outerCentroids,
    List<Color>   gradient,
    string        prefix,
    int           startIndex,
    ref object    labeledPanels,
    ref object    labelPoints,
    ref object    labelText,
    ref object    panelColors,
    ref object    names)
  {

=== STEP 1 -- DEFAULTS (before guards) ===

    // STEP 1 -- Defaults (before guards)
    List<Brep>    labeledBreps = new List<Brep>();
    List<Point3d> centroidList = new List<Point3d>();
    List<string>  textList     = new List<string>();
    List<Color>   colorList    = new List<Color>();
    List<string>  nameList     = new List<string>();

    if (prefix == null)    prefix = "";
    if (startIndex < 0)    startIndex = 0;

=== STEP 2 -- GUARDS ===

    // STEP 2 -- Guards
    if (panels == null || panels.Count == 0)
    {
      this.Component.AddRuntimeMessage(GH_RuntimeMessageLevel.Warning, "No panels connected");
      labeledPanels = labeledBreps;
      labelPoints   = centroidList;
      labelText     = textList;
      panelColors   = colorList;
      names         = nameList;
      return;
    }
    if (outerCentroids == null || outerCentroids.Count == 0)
    {
      this.Component.AddRuntimeMessage(GH_RuntimeMessageLevel.Warning, "No outerCentroids connected");
      labeledPanels = labeledBreps;
      labelPoints   = centroidList;
      labelText     = textList;
      panelColors   = colorList;
      names         = nameList;
      return;
    }
    if (gradient == null || gradient.Count < 2)
    {
      this.Component.AddRuntimeMessage(GH_RuntimeMessageLevel.Warning, "Connect at least 2 colors for gradient");
      labeledPanels = labeledBreps;
      labelPoints   = centroidList;
      labelText     = textList;
      panelColors   = colorList;
      names         = nameList;
      return;
    }

=== STEP 3 -- WORK LOOP ===

    // STEP 3 -- Per-panel loop
    int panelCount = panels.Count;
    for (int i = 0; i < panelCount; i++)
    {
      Brep brep = panels[i];
      if (brep == null) continue;

      // --- Name ---
      int    idx    = startIndex + i;
      string numStr = idx.ToString("D2");   // zero-pad to 2 digits minimum
      string name   = prefix + numStr;      // e.g. "P_01"

      // --- Embed name in UserDictionary ---
      Brep copy = brep.DuplicateBrep();
      copy.UserDictionary.Set("PartName",  name);
      copy.UserDictionary.Set("PartIndex", startIndex + i);
      labeledBreps.Add(copy);

      // --- Label point (pass-through from outerCentroids) ---
      Point3d centroid = (i < outerCentroids.Count) ? outerCentroids[i] : Point3d.Origin;
      centroidList.Add(centroid);

      // --- Label text ---
      textList.Add(name);
      nameList.Add(name);

      // --- Gradient color interpolation ---
      // t = normalized position of this panel along [0, 1]
      double t = (panelCount <= 1) ? 0.0 : (double)i / (panelCount - 1);

      // Map t to the gradient array (which may have 2, 3, or more stops)
      double scaledT     = t * (gradient.Count - 1);
      int    colorIdxLow = (int)Math.Floor(scaledT);
      int    colorIdxHigh= (int)Math.Ceiling(scaledT);
      if (colorIdxHigh >= gradient.Count) colorIdxHigh = gradient.Count - 1;
      double localT = scaledT - colorIdxLow;   // fractional part for lerp

      Color cLow  = gradient[colorIdxLow];
      Color cHigh = gradient[colorIdxHigh];

      int r = (int)(cLow.R + (cHigh.R - cLow.R) * localT);
      int g = (int)(cLow.G + (cHigh.G - cLow.G) * localT);
      int b = (int)(cLow.B + (cHigh.B - cLow.B) * localT);
      Color panelColor = Color.FromArgb(255, r, g, b);
      colorList.Add(panelColor);
    }

=== STEP 4 -- ASSIGN OUTPUTS (once, after loop) ===

    // STEP 4 -- Assign outputs (once, after loop)
    labeledPanels = labeledBreps;
    labelPoints   = centroidList;
    labelText     = textList;
    panelColors   = colorList;
    names         = nameList;
  }
}
