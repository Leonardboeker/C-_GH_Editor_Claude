// Template: Panel Volume Generator
// Inputs:  faces     -> List<Polyline>  (List Access)  -- one closed planar polyline per stone face
//          thickness -> double          (Item Access)  -- board thickness in mm (default 19.0)
// Outputs: panels        -> List<Brep>    -- one closed solid Brep per face, extruded outward
//          outerPlanes   -> List<Plane>   -- plane of the outer face per panel
//          outerCurves   -> List<Curve>   -- boundary curve of the outer face per panel
//          outerCentroids-> List<Point3d> -- centroid of the outer face per panel
//          report        -> List<string>  -- status message per panel


--- USINGS ---

using System;
using System.Collections;
using System.Collections.Generic;
using System.Drawing;

using Rhino;
using Rhino.Geometry;

using Grasshopper;
using Grasshopper.Kernel;
using Grasshopper.Kernel.Data;
using Grasshopper.Kernel.Types;


--- HELPER METHOD (same closed-safe pattern as miter_angle_calculator.cs) ---

private Point3d PolyCentroid(Polyline p)
{
  int n = p.Count;
  bool closed = (n > 1 && p[0].DistanceTo(p[n - 1]) < 1e-6);
  int verts = closed ? n - 1 : n;
  double sx = 0, sy = 0, sz = 0;
  for (int k = 0; k < verts; k++) { sx += p[k].X; sy += p[k].Y; sz += p[k].Z; }
  return new Point3d(sx / verts, sy / verts, sz / verts);
}


--- RunScript SIGNATURE ---

private void RunScript(
  List<Polyline> faces,
  double thickness,
  ref object panels,
  ref object outerPlanes,
  ref object outerCurves,
  ref object outerCentroids,
  ref object report)


--- ALGORITHM (implement exactly as described) ---

STEP 1 -- Defaults (before guards)
  panels        = new List<Brep>();
  outerPlanes   = new List<Plane>();
  outerCurves   = new List<Curve>();
  outerCentroids= new List<Point3d>();
  report        = new List<string>();

STEP 2 -- Guards
  if (faces == null || faces.Count == 0)
  {
    this.Component.AddRuntimeMessage(GH_RuntimeMessageLevel.Warning, "No face polylines connected");
    return;
  }
  if (thickness <= 0) thickness = 19.0;

STEP 3 -- Stone centroid approximation
  Compute the average of all face centroids (call PolyCentroid per face).
  This is the stone center used to determine outward direction.

  Point3d stoneCentroid = Point3d.Origin;  // initialize
  int validFaceCount = 0;
  for (int i = 0; i < faces.Count; i++)
  {
    if (faces[i] == null || faces[i].Count < 3) continue;
    Point3d fc = PolyCentroid(faces[i]);
    stoneCentroid += new Vector3d(fc);     // accumulate as vector
    validFaceCount++;
  }
  if (validFaceCount > 0)
    stoneCentroid = new Point3d(
      stoneCentroid.X / validFaceCount,
      stoneCentroid.Y / validFaceCount,
      stoneCentroid.Z / validFaceCount);

STEP 4 -- Per-face loop
  Declare result lists BEFORE the loop:
    List<Brep>    panelList    = new List<Brep>();
    List<Plane>   planeList    = new List<Plane>();
    List<Curve>   curveList    = new List<Curve>();
    List<Point3d> centroidList = new List<Point3d>();
    List<string>  reportList   = new List<string>();

  for (int i = 0; i < faces.Count; i++)
  {
    Polyline poly = faces[i];
    if (poly == null || poly.Count < 3)
    {
      reportList.Add($"Face {i}: skipped (null or degenerate)");
      continue;
    }

    // --- Normal direction (outward) ---
    // Use cross product of first two edges of the polyline
    Vector3d edge1 = poly[1] - poly[0];
    Vector3d edge2 = poly[2] - poly[1];
    Vector3d faceNormal = Vector3d.CrossProduct(edge1, edge2);
    if (!faceNormal.Unitize())
    {
      reportList.Add($"Face {i}: skipped (could not compute normal)");
      continue;
    }

    // Flip if normal points toward stone center instead of away
    Point3d faceCentroid = PolyCentroid(poly);
    Vector3d toCenter = stoneCentroid - faceCentroid;
    if (Vector3d.Multiply(toCenter, faceNormal) > 0)
    {
      faceNormal = -faceNormal;   // flip to point outward
    }

    // --- Extrusion ---
    Vector3d extrusionVec = faceNormal * thickness;
    NurbsCurve profileCrv = poly.ToNurbsCurve();
    Brep extruded = Brep.CreateFromExtrusion(profileCrv, extrusionVec);
    if (extruded == null)
    {
      reportList.Add($"Face {i}: extrusion failed");
      continue;
    }
    extruded.MakeValidForV2();   // ensure closed solid

    // --- Find outer face (largest area face = original polyline face) ---
    int outerFaceIndex = -1;
    double maxArea = -1.0;
    for (int k = 0; k < extruded.Faces.Count; k++)
    {
      AreaMassProperties amp = AreaMassProperties.Compute(extruded.Faces[k]);
      if (amp == null) continue;
      if (amp.Area > maxArea)
      {
        maxArea = amp.Area;
        outerFaceIndex = k;
      }
    }

    if (outerFaceIndex < 0)
    {
      reportList.Add($"Face {i}: could not identify outer face");
      continue;
    }

    BrepFace outerFace = extruded.Faces[outerFaceIndex];

    // --- Outer face plane (pre-declare, no out var) ---
    Plane outerFacePlane;
    bool planeOk = outerFace.TryGetPlane(out outerFacePlane);
    if (!planeOk) outerFacePlane = Plane.WorldXY;   // fallback

    // --- Outer face boundary curve ---
    Curve outerBoundaryCrv = outerFace.OuterLoop.To3dCurve();

    // --- Outer face centroid ---
    AreaMassProperties outerAmp = AreaMassProperties.Compute(outerFace);
    Point3d outerCentroid = (outerAmp != null) ? outerAmp.Centroid : faceCentroid;

    // Accumulate
    panelList.Add(extruded);
    planeList.Add(outerFacePlane);
    curveList.Add(outerBoundaryCrv);
    centroidList.Add(outerCentroid);
    reportList.Add($"Face {i}: OK  |  area: {maxArea:F1}  |  faces: {extruded.Faces.Count}");
  }

STEP 5 -- Assign outputs (once, after loop)
  panels         = panelList;
  outerPlanes    = planeList;
  outerCurves    = curveList;
  outerCentroids = centroidList;
  report         = reportList;
